Basics

# This is a starter program for Technology Days 
import turtle  # include all of the Turtle commands

tom = turtle.Turtle()  # names your Turtle - this Turtle's name is tom 
tom.shape("turtle")   # changes the arrow to a turtle shape

#tom.forward(n) statement to move the turtle forward n pixels
#tom.backward(n) statement to move the turtle backward n pixels

#tom.right(angle) statement to turn the turtle to the right by (angle) degrees
#tom.left(angle)  statement to turn the turtle to the left by (angle) degrees
#tom.setheading(angle) statement to set the turtle's heading

#tom.penup() statement to raise the pen up - don't draw
#tom.pendown() statement to lower the pen - start drawing

#tom.circle(radius) statement to draw a circle  .circle(100) is a complete circle

#tom.pensize(n)  statement to control the width of the line
#tom.pencolor(color)  statement to change the color of the pen
#tom.bgcolor(color) statement to set the background color of the screen
#tom.begin_fill(color) use this statement before drawing
#tom.end_fill() use this statement after drawing to complete filling the shape 

#tom.goto(x,y)  statement to move the turtle to a specific location 

turtle.done()

Spiral

import turtle

t = turtle.Turtle()     # names your Turtle - this Turtle's name is t 

# Add the statement to change the arrow to a turtle shape
# Add the statement to change the color of the turtle
 
NUM_CIRCLES =36         # the number of circles drawn 
RADIUS = 100            # radius of 100 is a complete circle
ANGLE = 10              # move to the left 10 degrees with each circle
ANIMATION_SPEED = 0     # how fast does your turtle draw 0 is fast 

t.speed(ANIMATION_SPEED)
 
for x in range(NUM_CIRCLES):       # this is a counter loop that will draw 36 circles
    t.circle(RADIUS)
    t.left(ANGLE)

turtle.done()
